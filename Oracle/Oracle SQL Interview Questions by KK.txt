-SQL query to find total salary department wise?

	FROM emp e INNER JOIN dept d ON e.deptno = d.deptno
	GROUP BY e.deptno,d.dname;
	
-Select ODD or EVEN rows from a table in Oracle?
	SELECT * FROM (SELECT empno,ename,sal,hiredate,ROWNUM RM)
	FROM emp ORDER BY empno)
	WHERE MOD(RM,2) = 0;
	
-SQL query to find total salary department wise?
	SELECT e.deptno,d.dname,SUM(e.sal) AS TOTAL_SALARY
	FROM emp e INNER JOIN dept d ON e.deptno = d.deptno
	GROUP BY e.deptno,d.dname;
	
-SQL query to find total salary department wise and sort total salary in ascending/descending order?
	SELECT e.deptno,d.dname,SUM(sal) AS total_salary
	FROM emp emp inner join dept d on e.deptno = d.deptno
	GROUP BY e.deptno,d.dname
	ORDER BY total_salary DESC;
	
-SQL query to find total salary department wise and their total salary is greater than given amount 
	SELECT e.deptno,d.dname,sum(sal)	
	FROM emp e inner join dept d on e.deptno = d.deptno
	GROUP BY e.deptno,d.dname
	HAVING SUM(sal) > 10000;
	
-SQL query to find total number of employees in each department ?
	select e.deptno,d.dname,count(*) "TOTAL_EMPLOYEES"
	from emp e INNER JOIN department d on e.deptno = d.deptno
	GROUP BY e.deptno,d.dname
	ORDER BY TOTAL_EMPLOYEES DESC;
	
-SQL to find departments whose number of employees is greater than the given count?
	SELECT e.deptno,d.dname,COUNT(*) AS  "TOTAL_EMPLOYEES"
	from emp e inner join dept d on e.deptno = d.deptno
	GROUP BY e.deptno,d.dname
	HAVING COUNT(*) > 4;
	
-SQL Query to find total employees department-wise and include department with zero employee
	SELECT d.dname,COUNT(e.deptno) "TOTAL_EMPLOYEES"
	FROM emp e RIGHT JOIN dept d ON e.deptno = d.deptno
	GROUP BY d.dname
	ORDER BY COUNT(e.deptno);
	
-SQL query to find average salary department wise ?
	SELECT d.deptno,d.dname,AVG(e.sal) "Average_Salary"
	FROM emp e RIGHT JOIN dept d on e.deptno = d.deptno
	GROUP BY d.deptno,d.dname
	ORDER BY AVG(sal);
-SQL Query to find 2nd or 3rd highest salary of employee?
	SELECT MAX(sal)
	FROM emp
	WHERE sal < (select max(sal) from emp);
-SQL Query to find Nth highest salary of employee ?
	SELECT * FROM emp e1
	WHERE N-1 = (
		SELECT COUNT(DISTINCT sal)
		FROM emp e2
		WHERE e2.sal > e1.sal
	);
-SQL Query to find Nth highest salary in Oracle using ROW_NUMBER function?
	DEFINE N =4;
	SELECT * FROM (SELECT e.*,ROW_NUMBER() OVER(
				ORDER BY sal DESC
			) AS rownumber FROM emp e )
	WHERE rownumber = &N;
-SQL Query to find Nth highest salary in Oracle using DENSE_RANK function ?
	SELECT * FROM (SELECT
		ename,sal, DENSE_RANK() OVER(
			ORDER BY SAL DESC
		) AS dr )
		FROM emp)
	WHERE dr = 3;
